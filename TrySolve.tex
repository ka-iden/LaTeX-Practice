    \[\int_{0}^{\frac{5}{2}\cos\theta}\pi\left(y\tan\theta + 1.5\right)^2dy\] \\
    \[= \pi\int_{0}^{\frac{5}{2}\cos\theta}\left(y\tan\theta + 1.5\right)^2dy\] \\
    \[= \pi\int_{0}^{\frac{5}{2}\cos\theta}\left(y\tan\theta + 1.5\right)\left(y\tan\theta + 1.5\right)dy\] \\
    \[= \pi\left[\left(y\tan\theta + 1.5\right)\left(\frac{\tan\theta}{2}y^2+1.5y\right)-\left(\tan\theta\right)\left(\frac{\tan\theta}{6}y^3+\frac{3}{4}y^2\right)\right]^{\frac{5}{2}\cos\theta}_0\] \\
    \[= \pi\left[\frac{1}{2}y^3\tan^2\theta+\frac{3}{2}y^2\tan\theta+\frac{3}{4}y^2\tan\theta+\frac{9}{4}y-\frac{1}{6}y^3\tan^2\theta+\frac{3}{4}y^3\tan\theta\right]^{\frac{5}{2}\cos\theta}_0\] \\
    \[= \pi\left[y^2\tan\theta\left(\frac{1}{2}y \tan\theta+\frac{3}{2}+\frac{3}{4}+\frac{9}{4}y-y^2\tan\theta\left(\frac{1}{6}y \tan\theta+\frac{3}{4}\right)\right)\right]^{\frac{5}{2}\cos\theta}_0\] \\
    \[= \pi\left[\frac{9}{4}y+y^2\tan\theta\left(\frac{1}{2}y\tan\theta+\frac{3}{2}+\frac{3}{2}-\frac{1}{6}y\tan\theta-\frac{3}{4}\right)\right]^{\frac{5}{2}\cos\theta}_0\] \\
    \[= \pi\left[\frac{9}{4}y+y^2\tan\theta\left(\frac{1}{3}y\tan\theta+\frac{3}{2}\right)\right]^{\frac{5}{2}\cos\theta}_0\] \\
    \[= \pi\left(\frac{9}{4}\left(\frac{5}{2}\cos\theta\right)+\frac{25}{4}\cos^{2}\theta\tan\theta\left(\frac{1}{3}\cdot\frac{5}{2}\cos\theta\tan\theta+\frac{3}{2}\right)\right)\] \\
    \[= \frac{5\pi}{8}\left(9\cos\theta+5\cos\theta\sin\theta\left(\frac{5}{3}\sin\theta+3\right)\right)\] \\
    \[= \frac{5\pi}{8}\cos\theta\left(9+5\sin\theta\left(\frac{5}{3}\sin\theta+3\right)\right)\] \\
